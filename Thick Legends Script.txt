getgenv().autoPunch = true -- Set to false if you wanna turn it off
getgenv().autoEat = true -- Set to false if you wanna turn it off
getgenv().autoSell = true -- Set to false if you wanna turn it off


spawn(function()
    while autoPunch == true do
        local args = {[1] = "Punch"}
        game:GetService("ReplicatedStorage").Remotes.UseTool:FireServer(unpack(args))
        wait()
    end
end)

spawn(function()
    while autoEat == true do
        local args = {[1] = "Food"}
        game:GetService("ReplicatedStorage").Remotes.UseTool:FireServer(unpack(args))
        wait()
    end
end)

spawn(function()
    while autoSell == true do
        game:GetService("ReplicatedStorage").Remotes.YieldSell:InvokeServer()
        wait()
    end
end)
